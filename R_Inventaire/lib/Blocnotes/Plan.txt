SELECT produits.id, produit.nom, types.nom, annee.valeur, fabricant.nom, provenance.pays, destination.pays, date_de_creation.date_production, lien.url, quantite.valeur, prix.montant, approbations.organisme, editeur.nom FROM `produits` JOIN produit ON produits.produit_id = produit.ID JOIN types ON produits.types_id = types.ID JOIN annee ON produits.annee_id = annee.ID JOIN fabricant ON produits.fabricant_id = fabricant.ID JOIN provenance ON produits.provenance_id = provenance.ID JOIN destination ON produits.destination_id = destination.ID JOIN date_de_creation ON produits.date_de_creation_id = date_de_creation.ID JOIN lien ON produits.lien_id = lien.ID JOIN quantite ON produits.quantite_id = quantite.ID JOIN prix ON produits.prix_id = prix.ID JOIN approbations ON produits.approbations_id = approbations.ID JOIN editeur ON produits.editeur_id = editeur.ID;
